    IS_NULLABLE,
    COLUMN_DEFAULT

-- 3. Verify foreign keys
SELECT
    fk.name                  AS FK_Name,
    tp.name                  AS ParentTable,
    cp.name                  AS ParentColumn,
    tr.name                  AS ReferencedTable,
    cr.name                  AS ReferencedColumn,
    fk.delete_referential_action_desc AS OnDelete
FROM   sys.foreign_keys       fk
JOIN   sys.tables             tp ON tp.object_id = fk.parent_object_id
JOIN   sys.tables             tr ON tr.object_id = fk.referenced_object_id
JOIN   sys.foreign_key_columns fkc ON fkc.constraint_object_id = fk.object_id
JOIN   sys.columns            cp  ON cp.object_id  = fkc.parent_object_id
                                  AND cp.column_id = fkc.parent_column_id
JOIN   sys.columns            cr  ON cr.object_id  = fkc.referenced_object_id
                                 AND cr.column_id  = fkc.referenced_column_id
ORDER  BY tp.name;

-- 4. Verify indexes
SELECT
    i.name           AS IndexName,
    t.name           AS TableName,
    c.name           AS ColumnName,
    i.is_unique      AS IsUnique
FROM   sys.indexes      i
JOIN   sys.tables       t  ON t.object_id = i.object_id
JOIN   sys.index_columns ic ON ic.object_id = i.object_id
                            AND ic.index_id  = i.index_id
JOIN   sys.columns      c  ON c.object_id  = ic.object_id
                            AND c.column_id = ic.column_id
WHERE  i.type > 0
  AND  t.name IN ('Projects','Tasks','TaskTimeLogs','Users')
ORDER  BY t.name, i.name;

Database Verification Query

-- View all time logs with task names
SELECT
    tl.Id,
    t.Title          AS TaskTitle,
    tl.StartTime     AT TIME ZONE 'UTC' AS StartUTC,
    tl.EndTime       AT TIME ZONE 'UTC' AS EndUTC,
    CASE
        WHEN tl.EndTime IS NULL THEN 'Running'
        ELSE CAST(
            ROUND(DATEDIFF(SECOND, tl.StartTime, tl.EndTime) / 3600.0, 4)
            AS VARCHAR) + 'h'
    END              AS Duration,
    tl.Notes,
    tl.IsDeleted
FROM TaskTimeLogs tl
JOIN Tasks t ON t.Id = tl.TaskId
WHERE tl.IsDeleted = 0
ORDER BY tl.StartTime DESC;

-- Total hours per task
SELECT
    t.Id,
    t.Title,
    COUNT(tl.Id)                                         AS Sessions,
    ROUND(SUM(
        DATEDIFF(SECOND, tl.StartTime, tl.EndTime)
    ) / 3600.0, 2)                                       AS TotalHours
FROM Tasks    t
JOIN TaskTimeLogs tl ON tl.TaskId    = t.Id
                     AND tl.EndTime  IS NOT NULL
                     AND tl.IsDeleted = 0
WHERE t.IsDeleted = 0
GROUP BY t.Id, t.Title
ORDER BY TotalHours DESC;
```

## Full Feature Flow
```
User clicks "Start Timer"
        │
        ▼
POST /TimeLog/Start
  TimeLogService.StartTimerAsync()
    ├── Check task exists
    ├── Check no active timer (prevent double start)
    ├── INSERT TaskTimeLog { StartTime = UtcNow, EndTime = null }
    └── Return log.Id + StartTime (ISO 8601)
        │
        ▼
timelog.js receives { success, startTime }
  ├── startLiveTick(startTime)  → updates #liveTimerDisplay every 1s
  ├── startLiveDurationTick()   → updates .live-duration in table
  └── loadSummary()             → refreshes partial with "Running" state

User clicks "Stop Timer"
        │
        ▼
POST /TimeLog/Stop
  TimeLogService.StopTimerAsync()
    ├── Find active log (EndTime IS NULL)
    ├── log.Stop() → EndTime = UtcNow (domain method)
    └── Return FormattedDuration
        │
        ▼
timelog.js receives { success, formattedDuration }
  ├── stopLiveTick()   → clears interval
  ├── showToast("Timer stopped. Duration: 1h 23m")
  └── loadSummary()    → refreshes partial with completed log
